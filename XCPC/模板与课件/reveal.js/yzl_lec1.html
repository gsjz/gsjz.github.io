void solve()
{
    auto fib = [&](this auto self, int n)
    { 
        if(n == 0) return 0;
        if(n == 1) return 1;
        return self(n-2) + self(n-1);
    };
    for(int i=0; i<=10; i++)
    {
        cout<<fib(i)<<"\n";
    }
    return;
}
